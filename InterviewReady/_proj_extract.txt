```java
public class FhirExternalClient {

    private final IGenericClient fhirClient;

    @CircuitBreaker(name = "externalFhir", fallbackMethod = "fetchFallback")
    // â†‘ Resilience4j: 5 consecutive fails â†’ circuit OPEN â†’ fail fast 60 sec

        // HAPI FHIR read â€” GET /DocumentReference/{id}
        return fhirClient.read()
            .resource(DocumentReference.class)
            .withId(documentId)
            .execute();
        // â†‘ Returns typed Java object â€” HAPI parses JSON automatically
    }

    public List<DocumentReference> searchByPatient(String patientId, String category) {
        // FHIR search: GET /DocumentReference?patient=123&category=lab
        return fhirClient.search()
            .forResource(DocumentReference.class)
            .where(DocumentReference.PATIENT.hasId(patientId))
            .and(DocumentReference.CATEGORY.exactly().code(category))
            .returnBundle(Bundle.class)
            .execute()
            .getEntry().stream()
            .map(e -> (DocumentReference) e.getResource())
            .toList();
    }

    DocumentReference fetchFallback(String documentId, Exception ex) {
        throw new FhirUnavailableException("External FHIR temporarily unavailable", ex);
    }
```

```java
@Target(ElementType.METHOD)
@Retention(RetentionPolicy.RUNTIME)
public @interface RequiresConsent {
    String scope();  // e.g. "VIEW_LAB_RESULTS", "AI_QUERY_ACCESS"
}

// â•â•â• Interceptor â€” har @RequiresConsent method pe auto-run â•â•â•
@Component
@RequiredArgsConstructor
public class ConsentInterceptor implements HandlerInterceptor {

    private final ConsentService consentService;
    private final AuditService auditService;

    @Override
    public boolean preHandle(HttpServletRequest request,
                             HttpServletResponse response,
                             Object handler) {
        if (!(handler instanceof HandlerMethod method)) return true;

        RequiresConsent annotation = method.getMethodAnnotation(RequiresConsent.class);
        if (annotation == null) return true;  // no consent required on this endpoint

        String patientId = request.getParameter("patientId");
        String requesterId = extractUserIdFromJwt(request);
        // â†‘ JWT sub claim = logged-in user ID

        ConsentCheckResponse result = consentService.check(
            patientId, annotation.scope(), requesterId
        );

        if (!result.isAllowed()) {
            auditService.log(request, patientId, requesterId,
                annotation.scope(), "DENIED", result.getReason());
            response.setStatus(HttpStatus.FORBIDDEN.value());
            // â†‘ 403 â€” fail-closed, NO PHI returned
            return false;  // stop request â€” controller method NOT called
        }

        auditService.log(request, patientId, requesterId,
            annotation.scope(), "ALLOWED", result.getConsentId());
        return true;  // proceed to controller
    }
}
```

Q: Consent REVOKE flow â€” race condition handling
â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€

REVOKE FLOW:
  1. DELETE /api/v1/consents/c-001 (patient JWT must match patientId)
  2. @Transactional: consent_db status=REVOKED + outbox event (SAME TX)
  3. Outbox worker â†’ Kafka consent-revoked { eventId, patientId, granteeId, scopes }
  4. Consumers react:
     fhir-sync: cancel in-progress syncs, evict Redis consent cache
     ai-assistant: clear session context, invalidate cached embeddings access
     notification: audit alert to compliance team

RACE CONDITION â€” revoke during in-flight FHIR fetch:
  Layer 1: Consent check at request START
  Layer 2: Consent RE-check before response RETURN to client
  Layer 3: Kafka revoke event â†’ abort in-progress fetch, discard partial data
  Layer 4: Access tokens tied to consentId â€” revoke = token invalid

â†’ Fail-closed: doubt ho to deny. HIPAA me unverified access > temporary block.


â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€

â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€

```java
public class ConsentCacheService {

    private static final Duration TTL = Duration.ofSeconds(60);
    // â†‘ 60 sec TTL â€” balance: performance vs revoke propagation speed
    // â†‘ Revoke pe Kafka event â†’ immediate cache evict (not wait for TTL)

    public ConsentCheckResponse check(String patientId, String granteeId, String scope) {
        String key = String.format("consent:active:%s:%s:%s", patientId, granteeId, scope);

        // STEP 1: Redis ElastiCache (~2ms) â€” 85ms â†’ 12ms improvement
        ConsentCheckResponse cached = redisTemplate.opsForValue().get(key);
        if (cached != null) return cached;

        // STEP 2: Cache miss â€” RDS query (~80ms)

            patientId, granteeId, scope);
```


SOLUTION (outbox pattern):
```java
// â•â•â• Outbox Entity â•â•â•
@Entity
@Table(name = "outbox_events")
public class OutboxEvent {
    @Id
    private UUID id;
    private String eventType;      // consent-revoked, document-received
    private String payload;        // JSON â€” IDs only, NO PHI in payload
    private boolean published;     // false until Kafka confirms
    private Instant createdAt;
}

// â•â•â• Same-transaction save â•â•â•
@Service
@RequiredArgsConstructor
public class ConsentService {

    @Transactional  // â† SINGLE transaction â€” both or neither
    public void revokeConsent(String consentId) {
        Consent consent = consentRepo.findById(consentId)
            .orElseThrow(() -> new NotFoundException(consentId));

        consent.setStatus(ConsentStatus.REVOKED);
        consentRepo.save(consent);

        // Outbox in SAME transaction â€” atomic with DB update
        outboxRepo.save(OutboxEvent.builder()
            .eventType("consent-revoked")
            .payload(toJson(Map.of(
                "eventId", UUID.randomUUID().toString(),
                "patientId", consent.getPatientId(),
                "granteeId", consent.getGranteeId(),
                "scopes", consent.getScopes()
                // â†‘ IDs only â€” NO patient name, lab results in Kafka payload
            )))
            .published(false)
            .build());
    }
}

// â•â•â• Outbox Worker â€” polls every 1 second â•â•â•
@Component
@Scheduled(fixedDelay = 1000)
public class OutboxPublisher {

    @Transactional
    public void publishPendingEvents() {
        List<OutboxEvent> pending = outboxRepo.findByPublishedFalse(
            PageRequest.of(0, 100));

        for (OutboxEvent event : pending) {
            kafkaTemplate.send(mapTopic(event.getEventType()), event.getPayload())
                .whenComplete((result, ex) -> {
                    if (ex == null) {
                        event.setPublished(true);
                        outboxRepo.save(event);
                    }
                    // â†‘ Fail â†’ stays unpublished â†’ retry next poll (1 sec)
                });
        }
    }
}
```

GUARANTEE: DB committed = event saved in outbox = eventually published to Kafka.

```java
@Component
@RequiredArgsConstructor
public class ConsentEventConsumer {

    private final ProcessedEventRepository processedEventRepo;
    private final ConsentCacheService cacheService;
    private final SyncJobService syncJobService;

    @KafkaListener(
        topics = "consent-events",
        groupId = "fhir-sync-group",
        // â†‘ groupId = consumer group â€” partitions shared among group members
        containerFactory = "kafkaListenerContainerFactory"
    )
    @Transactional
    public void handleConsentEvent(
            @Payload String payload,
            @Header(KafkaHeaders.RECEIVED_KEY) String key,
            Acknowledgment ack) {
        // â†‘ manual ack â€” commit offset ONLY after successful processing

        ConsentEvent event = objectMapper.readValue(payload, ConsentEvent.class);

        // IDEMPOTENCY CHECK â€” same event 3x process = process once
        if (processedEventRepo.existsByEventId(event.getEventId())) {
            ack.acknowledge();  // already processed â€” skip, commit offset
            return;
        }

        switch (event.getEventType()) {
            case "consent-revoked" -> {
                cacheService.evictAllForGrantee(event.getPatientId(), event.getGranteeId());
                syncJobService.cancelInProgressSyncs(event.getPatientId(), event.getGranteeId());
            }
            case "consent-granted" -> {
                cacheService.evictAllForGrantee(event.getPatientId(), event.getGranteeId());
                // â†‘ evict so next check hits DB with fresh consent
            }
        }

        processedEventRepo.save(new ProcessedEvent(event.getEventId()));
        ack.acknowledge();  // NOW commit offset â€” safe to mark done
    }
}
```

```java
@Service
@RequiredArgsConstructor
public class AiQueryOrchestrator {

    private final ConsentClient consentClient;
    private final EmbeddingService embeddingService;   // OpenAI embeddings
    private final VectorSearchService vectorSearch;    // pgvector
    private final ChatService chatService;             // OpenAI chat
    private final ResponseValidator validator;
    private final AuditService auditService;

    public AiQueryResponse processQuery(AiQueryRequest request, String requesterId) {

        // STEP 1: Consent check â€” BEFORE any OpenAI call (cost + HIPAA)
        ConsentCheckResponse consent = consentClient.check(
            request.getPatientId(), "AI_QUERY_ACCESS", requesterId);
        if (!consent.isAllowed()) {
            auditService.logUnauthorizedAiQuery(request.getPatientId(), requesterId);
            throw new ConsentDeniedException("AI access denied");
        }

        // STEP 2: Generate query embedding
        // "What was my blood sugar?" â†’ float[1536]

            request.getPatientId(), queryVector, 5);

            return AiQueryResponse.noResultsFound();
        }

            relevantChunks,    // retrieved context
            request.getQuery() // user question
        );

        // STEP 5: OpenAI chat completion (gpt-4, temperature=0.3)
        String rawAnswer = chatService.complete(prompt);

        // STEP 6: Validation â€” block diagnosis, require source citation
        ValidationResult validation = validator.validate(rawAnswer, relevantChunks);
        if (!validation.isValid()) {
            return AiQueryResponse.fallback(validation.getReason());
        }

        // STEP 7: Audit + response
        auditService.logAiQuery(request.getPatientId(), requesterId,
            relevantChunks.stream().map(DocumentChunk::getDocumentId).toList());

        return AiQueryResponse.builder()
            .answer(rawAnswer)
            .sources(mapSources(relevantChunks))
            .disclaimer("This is not medical advice. Consult your doctor.")
            .build();
    }
}
```

pgvector search SQL (annotated):
```sql
-- â•â•â• Semantic search query â•â•â•
SELECT
    chunk_text,                          -- actual text chunk from document
    document_id,                         -- source document reference
    document_type,
    1 - (embedding <=> $1::vector) AS similarity
    -- â†‘ <=> = cosine DISTANCE operator (pgvector)
    -- â†‘ 1 - distance = similarity score (0 to 1, higher = more similar)
    -- â†‘ $1 = query embedding vector[1536] from OpenAI
FROM document_embeddings
WHERE patient_id = $2                    -- consent scope filter
ORDER BY embedding <=> $1::vector        -- nearest neighbors first
LIMIT 5;                                 -- top-K retrieval for RAG context

-- IVFFlat index (2800ms â†’ 1900ms improvement):
CREATE INDEX idx_embeddings_vector ON document_embeddings
    USING ivfflat (embedding vector_cosine_ops) WITH (lists = 100);
-- â†‘ IVFFlat = approximate nearest neighbor â€” faster than exact scan
-- â†‘ lists = 100 clusters â€” tune based on row count (~sqrt(rows) rule)
```

```java
public class PromptBuilder {

    private static final String SYSTEM_PROMPT = """
        You are a healthcare document assistant. STRICT RULES:
        1. Answer ONLY from the provided document context below.
        2. NEVER diagnose conditions or recommend treatments.
        3. ALWAYS cite the source document ID for every fact.
        4. If information is not in context, say: "Information not found in your records."
        5. ALWAYS end with: "Please consult your doctor for medical advice."
        """;

    public String build(List<DocumentChunk> chunks, String userQuery) {
        StringBuilder context = new StringBuilder("DOCUMENT CONTEXT:\n");
        for (DocumentChunk chunk : chunks) {
            context.append(String.format(
                "[Doc: %s, Type: %s, Date: %s]\n%s\n\n",
                chunk.getDocumentId(),
                chunk.getDocumentType(),
                chunk.getDocumentDate(),
                chunk.getChunkText()
            ));
        }
        // SYSTEM (fixed rules) + CONTEXT (dynamic per query) + USER QUERY
        return SYSTEM_PROMPT + "\n" + context + "\nUSER QUESTION: " + userQuery;
    }
}
```

```java
public void indexDocument(@Payload DocumentReceivedEvent event, Acknowledgment ack) {

    if (processedEventRepo.existsByEventId(event.getEventId())) {
        ack.acknowledge(); return;
    }

    // STEP 1: Stream from S3 (NOT full 50MB in memory)

                .key(event.getS3Key())
                .build())) {

        String fullText = textExtractor.extract(docStream, event.getContentType());

        // STEP 3: Chunk â€” 500 tokens, 50 overlap

                .documentId(event.getDocumentId())
                .chunkIndex(i)
                .chunkText(chunks.get(i))
                .embedding(embedding)  // vector(1536) column
                .build());
        }
    }

    processedEventRepo.save(new ProcessedEvent(event.getEventId()));

}
```

```java
@KafkaListener(topics = "materialization-trigger", groupId = "materialization-group")
public void materialize(@Payload MaterializationEvent event, Acknowledgment ack) {

    if (processedEventRepo.existsByEventId(event.getEventId())) {
        ack.acknowledge(); return;
    }

    String warehouseId = event.getWarehouseId();

    // STEP 1: Read raw inventory data (indexed query â€” not full table scan)
    List<InventoryItem> items = inventoryRepo.findByWarehouseId(warehouseId);

    // STEP 2: Compute aggregates
    InventorySnapshot snapshot = InventorySnapshot.builder()
        .warehouseId(warehouseId)
        .totalSkus(items.size())
        .lowStockCount(items.stream().filter(i -> i.getQty() < i.getReorderLevel()).count())
        .totalValue(items.stream().mapToDouble(i -> i.getQty() * i.getUnitPrice()).sum())
        .computedAt(Instant.now())
        .build();

    // STEP 3: Write materialized table (upsert â€” idempotent)
    snapshotRepo.upsert(snapshot);

    // STEP 4: Update Redis cache â€” read path becomes ~50ms
    redisTemplate.opsForValue().set(
        "snapshot:" + warehouseId,
        snapshot,
        Duration.ofMinutes(5)  // TTL 5 min â€” stale max 5 min acceptable
    );

    processedEventRepo.save(new ProcessedEvent(event.getEventId()));
    ack.acknowledge();
}
```
